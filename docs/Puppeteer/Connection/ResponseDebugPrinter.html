def handle_message(message)
  if skip_debug_print?(message['method'])
    debug_print '.'
    @prev_log_skipped = true
  else
    debug_print "\n" if @prev_log_skipped
    @prev_log_skipped = nil
    debug_puts "RECV << #{decorate(message)}"
  end
end
